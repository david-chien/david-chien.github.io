def main():
    from sys import stdin, stdout
    e = stdin.readline
    w = stdout.write
    n, m = map(int, e().split())
    Y = [0] * n         # 這要記錄 Y 軸是否有值
    X = 0               # 這要記錄 X 軸
    for r in range(n):
        s = e()
        if('#' in s):   # 根據 readline
            Y[r] = 1    # 如果本列包含 #, Y[r] = 1
            # 這裡要記錄 X
            # 將 # 轉為 1 將 . 轉為 0 視為 2 進位字串
            # 將字串轉為整數，跟 X 做 or 運算，很快喔
            s = s.replace('#', '1').replace('.', '0')
            X |= int(s, 2)

    # 輸出就2種可能 Y[r] == 1, 整列輸出 mp
    # 否則輸出 m0
    # m0 先將 X 轉 2 進位，前面補 0
    # 將 1 轉 #, 將 0 轉 X
    mp = '#' * m + '\n'
    m0 = bin(X)[2:].zfill(m).replace('1', '#').replace('0', 'X') + '\n'
    for r in range(n):
        if(Y[r]): w(mp)
        else: w(m0)
main()
