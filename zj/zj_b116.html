def solve(arr, g):
    a = 1
    for x in arr: a |= (a << x)
    a = len(bin(a)[2:][:g+1].strip('0')) - 1
    return a

def main():
    from sys import stdin
    e = stdin.readline
    for s in stdin:
        m, n = map(int, s.split())
        for i in range(m):
            f = [int(x) for x in e().split()]
            sf = sum(f)
            if(sf & 1): print('No')
            else: 
                sf //= 2
                r = solve(f, sf)
                print('Yes' if r == sf else 'No')
main()
